#include <stdio.h>
void selectionSort(int arr[], int n) {
    for (int i = 0; i < n - 1; i++) {
        int Minindex = i;
        for (int j = i + 1; j < n; j++) {
            if (arr[j] < arr[Minindex]) {
                Minindex = j;
            }
        }
        int temp = arr[Minindex];
        arr[Minindex] = arr[i];
        arr[i] = temp;
    }
}
void printArray(int arr[], int n) {
    for (int i = 0; i < n; i++) {
        printf("%d ", arr[i]);
    }
    printf("\n");
}
int main() {
    int c;
    printf("Give the number of elements you want: ");
    scanf("%d", &c);
    int arr[c];
    printf("ENTER %d NUMBERS: \n", c);
    for (int i = 0; i < c; i++) {
        scanf("%d", &arr[i]);
    }
    selectionSort(arr, c);
    printf("Sorted array: ");
    printArray(arr, c);
    return 0;
}
